def convert(record)
  super { |field, struct|
    strings = []

    struct[:elements].each { |element|
      values = (struct[:values][element] || []).map { |v|
        (v || '').strip
      }.reject { |v| v.empty? }

      strings << (values.empty? ? struct[:empty] : values.join(VALUE_SEPARATOR))
    }

    builder.tag!(field, struct[:string] % strings)
  }
end
