def error(message)
  if @input.eos?
    raise "Unexpected end of input (#{message})."
  else
    raise "#{message} at #{$.}:#{@input.pos}: #{@input.peek(16).inspect}"
  end
end

def parse(input)
  @input = StringScanner.new(input)

  rows, block_given = [], block_given?

  while result = parse_row
    row = result.value
    block_given ? yield(row) : rows << row
    break unless @input.scan(%r,\s*/)
  end

  @input.scan(%r;/)  # optional

  error('Unexpected data') unless @input.eos?

  rows unless block_given
end

def parse_row
  return unless @input.scan(%r\(/)

  row = []

  while result = parse_value
    row << result.value
    break unless @input.scan(%r,\s*/)
  end

  error('Unclosed row') unless @input.scan(%r\)/)

  AST.new(row)
end

def parse_string
  return unless @input.scan(%r'/)

  string = ''

  while contents = parse_string_content || parse_string_escape
    string << contents.value
  end

  error('Unclosed string') unless @input.scan(%r'/)

  AST.new(string)
end

def parse_string_escape
  if @input.scan(%r\\[abtnvfr]/)
    AST.new(eval(%Q{"#{@input.matched}"}))
  elsif @input.scan(%r\\.|''/)
    AST.new(@input.matched[-1, 1])
  end
end
